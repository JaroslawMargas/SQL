CREATE TABLE Klienci(
    	IDKlient INT NOT NULL,
    	NazwaKlienta varchar(255) NOT NUL,
    	Andres varchar(255),
	KodPocztowy char(6),
	Miasto char(50),
    	Wojewodztwo char(50),
    CONSTRAINT PK_Klienci PRIMARY KEY (IDKlient)
);


CREATE TABLE Zamowienia(
    	NumerZamowienia INT NOT NULL,
	IDKlienta INT NOT NULL,
    	DataZamowienia DATE NOT NULL,
	WartZamNetto decimal (9,2),
	VAT decimal (9,2)
    CONSTRAINT PK_Zamowienia PRIMARY KEY (NumerZamowienia )
);
